94    public void testSingleRow() throws Exception {
95      ZooKeeperInstance zk = new ZooKeeperInstance(mac.getInstanceName(), mac.getZooKeepers());
96      Connector c = zk.getConnector("root", new PasswordToken("root"));
97      
98      final String tableName = "foo";
99      
100     if (c.tableOperations().exists(tableName)) {
101       c.tableOperations().delete(tableName);
102     }
103     
104     c.tableOperations().create(tableName);
105     
106     BatchWriter bw = null;
107     try {
108       bw = c.createBatchWriter(tableName, new BatchWriterConfig());
109       
110       Mutation m = new Mutation("1_a");
111       m.put("a", "a", 0, "");
112       m.put("a", "b", 0, "");
113       
114       bw.addMutation(m);
115       
116       m = new Mutation("1_b");
117       m.put("a", "a", 0, "");
118       
119       bw.addMutation(m);
120       
121       m = new Mutation("1_c");
122       m.put("a", "a", 0, "");
123       m.put("b", "a", 0, "");
124       m.put("c", "a", 0, "");
125       
126       bw.addMutation(m);
127     } finally {
128       if (null != bw) {
129         bw.close();
130       }
131     }
132     
133     Map<Key,Long> results = ImmutableMap.<Key,Long> builder().put(new Key("1_a", "a", "b", 0), 2l).put(new Key("1_b", "a", "a", 0), 1l)
134         .put(new Key("1_c", "c", "a", 0), 3l).build();
135     
136     BatchScanner bs = c.createBatchScanner(tableName, new Authorizations(), 1);
137     bs.setRanges(Collections.singleton(new Range()));
138     
139     try {
140       IteratorSetting cfg = new IteratorSetting(30, GroupByRowSuffixIterator.class);
141       
142       bs.addScanIterator(cfg);
143       
144       long count = 0;
145       for (Entry<Key,Value> entry : bs) {
146         VLongWritable writable = GroupByRowSuffixIterator.getWritable(entry.getValue());
147         
148         Assert.assertTrue("Results did not contain: " + entry.getKey(), results.containsKey(entry.getKey()));
149         Assert.assertEquals(results.get(entry.getKey()).longValue(), writable.get());
150         
151         count++;
152       }
153       
154       Assert.assertEquals(results.size(), count);
155     } finally {
156       if (null != bs) {
157         bs.close();
158       }
159     }

379   @Test
380   public void testReseek() throws Exception {
381     TreeMap<Key,Value> data = Maps.newTreeMap();
382     data.put(new Key("foo\u0000bell", "RESTAURANT", "f\u00001"), new Value());
383     data.put(new Key("foo\u0000bell", "RESTAURANT", "f\u00002"), new Value());
384     data.put(new Key("foo\u0000taco", "RESTAURANT", "f\u00001"), new Value());
385     data.put(new Key("foo\u0000taco", "RESTAURANT", "f\u00002"), new Value());
386     data.put(new Key("foo\u0000taco", "RESTAURANT", "f\u00003"), new Value());
387     
388     SortedMapIterator source = new SortedMapIterator(data);
389     
390     GroupByRowSuffixIterator iter = new GroupByRowSuffixIterator();
391     iter.init(source, Collections.<String,String> emptyMap(), new IteratorEnvironment() {
392       @Override
393       public SortedKeyValueIterator<Key,Value> reserveMapFileReader(final String mapFileName) throws IOException {
394         return null;
395       }
396       
397       @Override
398       public AccumuloConfiguration getConfig() {
399         return null;
400       }
401       
402       @Override
403       public IteratorScope getIteratorScope() {
404         return null;
405       }
406       
407       @Override
408       public boolean isFullMajorCompaction() {
409         return false;
410       }
411       
412       @Override
413       public void registerSideChannel(final SortedKeyValueIterator<Key,Value> iter) {}
414     });
415     
416     iter.seek(Range.prefix("foo"), Collections.<ByteSequence> emptySet(), false);
417     
418     Assert.assertTrue(iter.hasTop());
419     
420     Assert.assertEquals(new Key("foo\u0000bell", "RESTAURANT", "f\u00002"), iter.getTopKey());
421     VLongWritable actual = new VLongWritable(), expected = new VLongWritable(2);
422     
423     actual.readFields(new DataInputStream(new ByteArrayInputStream(iter.getTopValue().get())));
424     Assert.assertEquals(expected, actual);
425     
426     iter.seek(new Range(new Key("foo\u0000bell", "RESTAURANT", "f\u00002"), false, new Key("fop"), false), Collections.<ByteSequence> emptySet(), false);
427     
428     Assert.assertTrue(iter.hasTop());
429 
430     Assert.assertEquals(new Key("foo\u0000taco", "RESTAURANT", "f\u00003"), iter.getTopKey());
431     actual = new VLongWritable();
432     expected = new VLongWritable(3);
433     
434     actual.readFields(new DataInputStream(new ByteArrayInputStream(iter.getTopValue().get())));
435     Assert.assertEquals(expected, actual);
436     
437     iter.next();
438     
439     Assert.assertFalse(iter.hasTop());
440   }
