36  public class MultimapQueryResultTest extends AbstractSortableTest {
37    
38    @Test
39    public void identityWritableEquality() throws Exception {
40      Multimap<Column,SValue> data = HashMultimap.create();
41      
42      data.put(Column.create("TEXT"), SValue.create("foo", VIZ));
43      data.put(Column.create("TEXT"), SValue.create("bar", VIZ));
44      
45      MultimapQueryResult mqr = new MultimapQueryResult(data, "1", VIZ);
46      
47      DataOutputBuffer out = new DataOutputBuffer();
48      mqr.write(out);
49      
50      DataInputBuffer in = new DataInputBuffer();
51      
52      byte[] bytes = out.getData();
53      in.reset(bytes, out.getLength());
54      
55      MultimapQueryResult mqr2 = MultimapQueryResult.recreate(in);
56      
57      Assert.assertEquals(mqr, mqr2);
58    }
59    
60    @Test
61    public void nonEqual() throws Exception {
62      Multimap<Column,SValue> data = HashMultimap.create();
63      
64      data.put(Column.create("TEXT"), SValue.create("foo", VIZ));
65      data.put(Column.create("TEXT"), SValue.create("bar", VIZ));
66      
67      MultimapQueryResult mqr = new MultimapQueryResult(data, "1", VIZ);
68      MultimapQueryResult mqr2 = new MultimapQueryResult(data, "2", VIZ);
69      
70      Assert.assertNotEquals(mqr, mqr2);
71      
72      MultimapQueryResult mqr3 = new MultimapQueryResult(data, "1", new ColumnVisibility("foobarbarbarbarbarbar"));
73      
74      Assert.assertNotEquals(mqr, mqr3);
75      Assert.assertNotEquals(mqr2, mqr3);
76      
77      data = HashMultimap.create(data);
78      
79      data.put(Column.create("FOO"), SValue.create("barfoo", VIZ));
80      
81      MultimapQueryResult mqr4 = new MultimapQueryResult(data, "1", VIZ);
82      
83      Assert.assertNotEquals(mqr, mqr4);
84      Assert.assertNotEquals(mqr2, mqr4);
85      Assert.assertNotEquals(mqr3, mqr4);
86    }
