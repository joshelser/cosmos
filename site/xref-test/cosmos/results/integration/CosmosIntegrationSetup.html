215   @SuppressWarnings({"rawtypes", "unchecked"})
216   protected static MediaWikiType loadWiki(int num) throws Exception {
217     Unmarshaller unmarshaller = context.createUnmarshaller();
218     
219     InputStream is = CosmosIntegrationSetup.class.getResourceAsStream(ARTICLE_BASE + num + ARTICLE_SUFFIX);
220     
221     Assert.assertNotNull(is);
222     
223     GZIPInputStream gzip = new GZIPInputStream(is);
224     Object o = unmarshaller.unmarshal(gzip);
225     
226     Assert.assertEquals(JAXBElement.class, o.getClass());
227     Assert.assertEquals(MediaWikiType.class, ((JAXBElement) o).getDeclaredType());
228     
229     JAXBElement<MediaWikiType> jaxb = (JAXBElement<MediaWikiType>) o;
230     
231     return jaxb.getValue();
232   }

234   public static List<QueryResult<?>> wikiToMultimap(MediaWikiType wiki) {
235     Preconditions.checkNotNull(wiki);
236     
237     List<PageType> pages = wiki.getPage();
238     List<QueryResult<?>> mmap = Lists.newArrayList();
239     final String lang = wiki.getLang();
240     final ColumnVisibility viz = new ColumnVisibility(lang);
241     long id = 0l;
242     
243     for (PageType page : pages) {
244       Multimap<Column,SValue> data = HashMultimap.create();
245       
246       data.put(Column.create(PAGE_ID), SValue.create(page.getId().toString(), viz));
247       data.put(Column.create(PAGE_TITLE), SValue.create(page.getTitle(), viz));
248       
249       if (!StringUtils.isBlank(page.getRestrictions())) {
250         data.put(Column.create(PAGE_RESTRICTIONS), SValue.create(page.getRestrictions(), viz));
251       }
252       
253       List<Object> revisions = page.getRevisionOrUploadOrLogitem();
254       for (Object o : revisions) {
255         if (o instanceof RevisionType) {
256           RevisionType rev = (RevisionType) o;
257           
258           if (null != rev.getContributor()) {
259             // If we have an IP, not a logged in user
260             if (null != rev.getContributor().getIp()) {
261               data.put(Column.create(CONTRIBUTOR_IP), SValue.create(rev.getContributor().getIp(), viz));
262             } else {
263               // Assume username with ID
264               data.put(Column.create(CONTRIBUTOR_USERNAME), SValue.create(rev.getContributor().getUsername(), viz));
265               data.put(Column.create(CONTRIBUTOR_ID), SValue.create(rev.getContributor().getId().toString(), viz));
266             }
267           }
268           data.put(Column.create(REVISION_ID), SValue.create(rev.getId().toString(), viz));
269           data.put(Column.create(REVISION_TIMESTAMP), SValue.create(rev.getTimestamp().toString(), viz));
270           
271           if (null != rev.getComment() && !StringUtils.isBlank(rev.getComment().getValue())) {
272             data.put(Column.create(REVISION_COMMENT), SValue.create(rev.getComment().getValue(), viz));
273           }
274         }
275       }
276       
277       mmap.add(new MultimapQueryResult(data, lang + id, viz));
278       id++;
279     }
280     
281     return mmap;
282   }
