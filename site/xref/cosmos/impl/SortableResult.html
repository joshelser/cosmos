71    public SortableResult(Connector connector, Authorizations auths, Set<Index> columnsToIndex, boolean lockOnUpdates, String dataTable, String metadataTable) {
72      checkNotNull(connector);
73      checkNotNull(auths);
74      checkNotNull(columnsToIndex);
75      checkNotNull(dataTable);
76      checkNotNull(metadataTable);
77      
78      this.connector = connector;
79      this.auths = auths;
80      this.lockOnUpdates = lockOnUpdates;
81      
82      // Make sure we don't try to make a real Set out of the IdentitySet
83      if (columnsToIndex instanceof IdentitySet) {
84        this.columnsToIndex = columnsToIndex;
85      } else {
86        this.columnsToIndex = Sets.newHashSet(columnsToIndex);
87      }
88      
89      this.dataTable = dataTable;
90      this.metadataTable = metadataTable;
91      
92      this.UUID = randomUUID().toString();
93      
94      TableOperations tops = this.connector.tableOperations();
95      
96      createIfNotExists(tops, this.dataTable());
97      splitTable(tops, this.dataTable());
98      addLocalityGroups(tops, this.dataTable());
99      createIfNotExists(tops, this.metadataTable());
100   }

102   protected void createIfNotExists(TableOperations tops, String tableName) {
103     if (!tops.exists(tableName)) {
104       try {
105         tops.create(tableName);
106         
107         // TODO Make a better API than runtimeexception? Do (should) I care?
108         // If the user we were given can't do what's necessary, then
109         // it needed to be done ahead of time. Either way it's fatal?
110         // I suppose best to just make a named-exception then so people
111         // specifically know what happened.
112       } catch (AccumuloException e) {
113         log.error("Could not create table '{}'", tableName, e);
114         throw new RuntimeException(e);
115       } catch (AccumuloSecurityException e) {
116         log.error("Could not create table '{}'", tableName, e);
117         throw new RuntimeException(e);
118       } catch (TableExistsException e) {
119         log.error("Could not create table '{}'", tableName, e);
120         throw new RuntimeException(e);
121       }
122     }
123   }

125   /**
126    * Make sure we have a reasonable number of splits for the data table
127    * or else concurrency will just grind to a halt. 
128    * @param tops
129    * @param tableName
130    */
131   protected void splitTable(TableOperations tops, String tableName) {
132     try {
133       // Having 10 splits should be no problem on a single machine
134       // so we can always split to that point
135       final int EXPECTED_SPLITS = 10;
136       Collection<Text> splits = tops.listSplits(tableName, EXPECTED_SPLITS);
137       
138       if (splits.size() < EXPECTED_SPLITS) {
139         tops.addSplits(tableName, SPLITS);
140       }
141     } catch (TableNotFoundException e) {
142       log.error("Could not add splits to table '{}'", tableName, e);
143       throw new RuntimeException(e);
144     } catch (AccumuloException e) {
145       log.error("Could not add splits to table '{}'", tableName, e);
146       throw new RuntimeException(e);
147     } catch (AccumuloSecurityException e) {
148       log.error("Could not add splits to table '{}'", tableName, e);
149       throw new RuntimeException(e);
150     }
151   }

153   /**
154    * Ensure that the {@link Defaults.CONTENT_LG_NAME} locality group is configured
155    * @param tops
156    * @param tableName
157    */
158   protected void addLocalityGroups(TableOperations tops, String tableName) {
159     try {
160       Map<String,Set<Text>> localityGroups = tops.getLocalityGroups(tableName);
161       
162       // If we don't have a locality group specified with the expected name
163       // create one automatically
164       if (!localityGroups.containsKey(Defaults.CONTENTS_LG_NAME)) {
165         localityGroups.put(Defaults.CONTENTS_LG_NAME, Collections.singleton(Defaults.CONTENTS_COLFAM_TEXT));
166         
167         tops.setLocalityGroups(tableName, localityGroups);
168       } else {
169         Set<Text> colfams = localityGroups.get(Defaults.CONTENTS_LG_NAME);
170         if (!colfams.contains(Defaults.CONTENTS_COLFAM_TEXT)) {
171           log.warn("The {} locality group does not contain the expected column family {}", Defaults.CONTENTS_LG_NAME, Defaults.CONTENTS_COLFAM_TEXT);
172         }
173       }
174     } catch (AccumuloException e) {
175       log.error("Could not add locality groups to table '{}'", tableName, e);
176       throw new RuntimeException(e);
177     } catch (AccumuloSecurityException e) {
178       log.error("Could not add locality groups to table '{}'", tableName, e);
179       throw new RuntimeException(e);
180     } catch (TableNotFoundException e) {
181       log.error("Could not add locality groups to table '{}'", tableName, e);
182       throw new RuntimeException(e);
183     }
184   }

214   protected void addColumnsToIndex(Collection<Index> columns) {
215     checkNotNull(columns);
216     
217     if (IdentitySet.class.isAssignableFrom(columns.getClass())) {
218       // We got an IdentitySet, so we're now an IdentitySet
219       this.columnsToIndex = (IdentitySet<Index>) columns;
220     } else if (!(IdentitySet.class.isAssignableFrom(this.columnsToIndex.getClass()))) {
221       // We aren't already an IdentitySet
222       this.columnsToIndex.addAll(columns);
223     }
224   }
