22  import java.nio.charset.CharacterCodingException;

30  import com.google.common.base.Function;

32  import cosmos.Cosmos;
33  import cosmos.UnexpectedStateException;
34  import cosmos.options.Defaults;
35  import cosmos.results.impl.MultimapQueryResult;

37  /**
38   * 
39   */
40  public class IndexToMultimapQueryResult implements Function<Entry<Key,Value>,MultimapQueryResult> {
41    
42    protected final Cosmos sorts;
43    protected final SortableResult id;
44    
45    public IndexToMultimapQueryResult(Cosmos sorts, SortableResult id) {
46      this.sorts = sorts;
47      this.id = id;
48    }
49    
50    @Override
51    public MultimapQueryResult apply(Entry<Key,Value> input) {
52      Key k = input.getKey();
53      
54      Text colqual = k.getColumnQualifier();
55      
56      int index = colqual.find(Defaults.NULL_BYTE_STR);
57      if (-1 == index) {
58        throw new RuntimeException("Was provided unexpected Key: " + k);
59      }
60      
61      int start = index + 1;
62      try {
63        String docId = Text.decode(colqual.getBytes(), start, colqual.getLength() - start);
64        
65        return sorts.contents(id, docId);
66        
67      } catch (TableNotFoundException e) {
68        throw new RuntimeException(e);
69      } catch (UnexpectedStateException e) {
70        throw new RuntimeException(e);
71      } catch (CharacterCodingException e) {
72        throw new RuntimeException(e);
73      }
74    }
75    
76  }
