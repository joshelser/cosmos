49  public class CosmosTest {
50    public static void foo(String[] args) throws Exception {
51      List<String> data = Lists.newArrayList("aaa", "aab");
52      
53      StringLexicoder lex = new StringLexicoder();
54      ReverseLexicoder<String> revlex = new ReverseLexicoder<String>(lex);
55      
56      byte[] b1 = lex.encode(data.get(0)), b2 = lex.encode(data.get(1));
57      
58      System.out.println(new String(b1));
59      System.out.println(new String(b2));
60      
61      b1 = revlex.encode(data.get(0));
62      b2 = revlex.encode(data.get(1));
63  
64      List<String> revData = Lists.newArrayList(new String(b1), new String(b2));
65      Collections.sort(revData);
66      for (String s : revData) {
67        System.out.println(s);
68        System.out.println(Arrays.toString(s.getBytes()));
69      }
70      
71      System.exit(1);
72    }
73    
74    public static void main(String[] args) throws Exception {
75      foo(args);
76      ZooKeeperInstance zk = new ZooKeeperInstance("accumulo1.5", "localhost");
77      Connector con = zk.getConnector("mediawiki", new PasswordToken("password"));
78  
79      BatchScanner bs = con.createBatchScanner(Defaults.DATA_TABLE, con.securityOperations().getUserAuthorizations("mediawiki"), 10);
80      bs.setRanges(Collections.singleton(Range.prefix("55b672e0-a5c8-4ff7-81a9-bfd7e52cde74")));
81      bs.fetchColumnFamily(new Text("CONTRIBUTOR_USERNAME"));
82  
83  
84      // TODO Need to post filter on cq-prefix to only look at the ordering we want
85      IteratorSetting filter = new IteratorSetting(50, "cqFilter", OrderFilter.class);
86      filter.addOption(OrderFilter.PREFIX, Order.FORWARD);
87      bs.addScanIterator(filter);
88      
89      IteratorSetting cfg = new IteratorSetting(60, GroupByRowSuffixIterator.class);
90      bs.addScanIterator(cfg);
91      
92      CloseableIterable<Entry<SValue,Long>> iter = CloseableIterable.transform(bs, new GroupByFunction());
93      for (Entry<SValue,Long> entry : iter) {
94        System.out.println(entry.getKey().value() + "=" + entry.getValue());
95      }
96      
97      bs.close();
98    }
