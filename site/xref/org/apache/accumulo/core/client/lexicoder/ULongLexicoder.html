17  package org.apache.accumulo.core.client.lexicoder;
18  
19  /**
20   * Unsigned long lexicoder. The lexicographic encoding sorts first 0l and -1l last. This encoding does not correspond to the sort of Long because it does not
21   * consider the sign bit. If Java had an unsigned long type, this encoder would correspond to its sort order.
22   * 
23   * @since 1.6.0
24   */
25  public class ULongLexicoder implements Lexicoder<Long> {
26    
27    @Override
28    public byte[] encode(Long l) {
29      int shift = 56;
30      int index;
31      int prefix = l < 0 ? 0xff : 0x00;
32      
33      for (index = 0; index < 8; index++) {
34        if (((l >>> shift) & 0xff) != prefix)
35          break;
36        
37        shift -= 8;
38      }
39      
40      byte ret[] = new byte[9 - index];
41      ret[0] = (byte) (8 - index);
42      for (index = 1; index < ret.length; index++) {
43        ret[index] = (byte) (l >>> shift);
44        shift -= 8;
45      }
46      
47      if (l < 0)
48        ret[0] = (byte) (16 - ret[0]);
49      
50      return ret;
51      
52    }
53    
54    @Override
55    public Long decode(byte[] data) {
56      
57      long l = 0;
58      int shift = 0;
59      
60      if (data[0] < 0 || data[0] > 16)
61        throw new IllegalArgumentException("Unexpected length " + (0xff & data[0]));
62      
63      for (int i = data.length - 1; i >= 1; i--) {
64        l += (data[i] & 0xffl) << shift;
65        shift += 8;
66      }
67      
68      // fill in 0xff prefix
69      if (data[0] > 8)
70        l |= -1l << ((16 - data[0]) << 3);
71      
72      return l;
73    }
74  }
